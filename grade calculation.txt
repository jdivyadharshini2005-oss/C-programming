#include<stdio.h>
int main(){
    int n;
    scanf("%d",&n);
    if(n<100 && n>90){
        printf("grade A");
    }
    else if(n<90 && n>80){
        printf("grade B");
    }
    else if(n<80 && n>70){
        printf("grade C");
    }
    else if(n<70 && n>60){
        printf("grade D");
    }
    else {
        printf("grade E");
    }
}
