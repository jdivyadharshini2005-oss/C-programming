#include <stdio.h>
int main()
{
    int n, temp, digit, sum = 0, i, fact;
    scanf("%d", &n);
    temp = n;
    for (; temp > 0; temp = temp / 10)
    {
        digit = temp % 10;
        fact = 1;
        for (i = 1; i <= digit; i++)
        {
            fact = fact * i;
        }
        sum = sum + fact;
    }
    if (sum == n)
        printf("yes");
    else
        printf("No");

    return 0;
}
