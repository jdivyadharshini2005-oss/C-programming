#include <stdio.h>

int main()
{
    int n, temp, digit, sum = 0;

    scanf("%d", &n);
    temp = n;

    for (; temp > 0; temp = temp / 10)
    {
        digit = temp % 10;
        sum = sum + (digit * digit * digit);
    }

    if (sum == n)
        printf("Yes");
    else
        printf("No");

    return 0;
}
