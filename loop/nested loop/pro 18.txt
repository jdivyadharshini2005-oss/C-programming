ABCDCBA
#include <stdio.h>

int main() {
    int i, j;
    char ch;

    for (i = 0; i < 4; i++) {
        ch = 'A';

        // Increasing part
        for (j = 0; j <= i; j++) {
            printf("%c", ch);
            ch++;
        }

        // Decreasing part
        ch -= 2;
        for (j = 0; j < i; j++) {
            printf("%c", ch);
            ch--;
        }

        printf("\n");
    }
    return 0;
}
