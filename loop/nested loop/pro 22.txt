#include <stdio.h>

int main() {
    int i, j, n = 4;

    for (i = -n+1; i < n; i++) {
        for (j = -n+1; j < n; j++) {
            int val = n - (abs(i) > abs(j) ? abs(i) : abs(j));
            printf("%d ", val);
        }
        printf("\n");
    }

    return 0;
}
