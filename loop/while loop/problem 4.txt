#include <stdio.h>

int main() {
    int n, i = 0;
    int order;
    int success = 0, cancelled = 0;

    scanf("%d", &n);

    while (i < n) {
        scanf("%d", &order);

        if (order == 1) {
            success++;
        } else {
            cancelled++;
        }

        i++;
    }

    printf("Successful: %d\n", success);
    printf("Cancelled: %d\n", cancelled);

    if (cancelled > success) {
        printf("Status: Risk\n");
    } else {
        printf("Status: Safe\n");
    }

    return 0;
}
