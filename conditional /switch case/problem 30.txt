#include <stdio.h>
int main() {
    int hrs;
    scanf("%d", &hrs);

    switch(hrs) {
        case 1:
            printf("Short Interruption");
            break;

        case 2:
        case 3:
        case 4:
            printf("Medium Interruption");
            break;

        default:
            printf("Long Interruption");
    }
    return 0;
}
