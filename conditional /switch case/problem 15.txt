#include <stdio.h>

int main() {
    int mode;
    char cat;
    scanf("%d %c", &mode, &cat);

    switch(mode) {
        case 1:
            if(cat == 'R')
                printf("₹5000");
            else
                printf("₹3000");
            break;

        case 2:
            if(cat == 'R')
                printf("₹9000");
            else
                printf("₹7000");
            break;
    }
    return 0;
}
